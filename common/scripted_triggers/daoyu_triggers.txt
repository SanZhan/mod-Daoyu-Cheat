				has_variable = daoyu_target_state

			}

			has_global_variable = daoyu_add_cultural_obsession
			any_country = { leading_producer_of = g:$GOODS$ }
			or = {
				any_country = { has_variable = daoyu_target_country any_primary_culture = { not = { has_cultural_obsession = $GOODS$ } } }
				and = { has_variable = daoyu_mark_target_culture_obsession exists = scope:daoyu_mark_target_culture scope:daoyu_mark_target_culture = { not = { has_cultural_obsession = $GOODS$ } } }
			}
		}
		and = {
			has_global_variable = daoyu_remove_cultural_obsession
			or = {
				any_country = { has_variable = daoyu_target_country any_primary_culture = { has_cultural_obsession = $GOODS$ } }
				and = { has_variable = daoyu_mark_target_culture_obsession exists = scope:daoyu_mark_target_culture scope:daoyu_mark_target_culture = { has_cultural_obsession = $GOODS$ } }
			}
		}
		and = {
			has_global_variable = daoyu_add_banned_goods
			any_country = { leading_producer_of = g:$GOODS$ }
			any_country = { has_variable = daoyu_target_country is_banning_goods = g:$GOODS$ }
		}
		and = {
			has_global_variable = daoyu_remove_banned_goods
			any_country = { has_variable = daoyu_target_country is_banning_goods = g:$GOODS$ }
		}
		and = {
			has_global_variable = daoyu_increase_price_goods
			any_country = { has_variable = daoyu_target_country can_establish_any_export_route = g:$GOODS$ any_market = { daoyu_can_create_trade_route_trigger = { SCOPE = scope:daoyu_target_country } } }
		}
		and = {
			has_global_variable = daoyu_decrease_price_goods
			any_country = { has_variable = daoyu_target_country can_establish_any_import_route = g:$GOODS$ any_market = { daoyu_can_create_trade_route_trigger = { SCOPE = scope:daoyu_target_country } } }
		}
	}
}
