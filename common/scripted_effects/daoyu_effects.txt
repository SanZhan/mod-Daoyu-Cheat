daoyu_single_character_ideology = {
	if = {
		limit = { has_global_variable = daoyu_remove_show_character_ideology_effect }
		if = { limit = { has_variable = daoyu_show_CI_$IDEOLOGY$ } remove_variable = daoyu_show_CI_$IDEOLOGY$ }
	}
	if = {
		limit = { has_global_variable = daoyu_count_character_ideology_effect }
		if = {
			limit = {
				or = {
					and = { has_variable = daoyu_show_available any_character = { has_trait = daoyu_target_character not = { has_ideology = ideology:$IDEOLOGY$ } daoyu_$IDEOLOGY$ = yes } }
					and = { has_variable = daoyu_show_more any_character = { has_trait = daoyu_target_character not = { has_ideology = ideology:$IDEOLOGY$ } daoyu_$IDEOLOGY$ = no } }
				}
			}
			change_variable = { name = daoyu_count add = 1 }
			if = { limit = { has_variable = daoyu_page_01 has_variable = daoyu_count var:daoyu_count <= 14 } set_variable = daoyu_show_CI_$IDEOLOGY$ }
			if = { limit = { has_variable = daoyu_page_02 has_variable = daoyu_count var:daoyu_count > 14 var:daoyu_count <= 27 } set_variable = daoyu_show_CI_$IDEOLOGY$ }
			if = { limit = { has_variable = daoyu_page_03 has_variable = daoyu_count var:daoyu_count > 27 var:daoyu_count <= 40 } set_variable = daoyu_show_CI_$IDEOLOGY$ }
			if = { limit = { has_variable = daoyu_page_04 has_variable = daoyu_count var:daoyu_count > 40 } set_variable = daoyu_show_CI_$IDEOLOGY$ }
		}
	}
}

daoyu_create_character = {
	random_country = {
		limit = { has_variable = daoyu_target_country }
		if = {
			limit = { has_global_variable = daoyu_create_general }
			create_character = { age = 20 is_general = yes ig_leader = no noble = no interest_group = $IG$ traits = { daoyu_target_character wrathful } }
		}
		else_if = {
			limit = { has_global_variable = daoyu_create_admiral }
			create_character = { age = 20 is_admiral = yes ig_leader = no noble = no interest_group = $IG$ traits = { daoyu_target_character wrathful } }
		}
		else_if = {
			limit = { has_global_variable = daoyu_create_character_male }
			if = {
				limit = { has_global_variable = daoyu_create_politician }
				if = { limit = { has_global_variable = daoyu_noble } create_character = { age = 35 ig_leader = yes noble = yes interest_group = $IG$ traits = { daoyu_target_character } } }
				else = { create_character = { age = 35 ig_leader = yes noble = no interest_group = $IG$ traits = { daoyu_target_character } } }
			}
			else_if = {
				limit = { has_global_variable = daoyu_create_agitator }
				if = { limit = { has_global_variable = daoyu_noble } create_character = { age = 35 is_agitator = yes noble = yes interest_group = $IG$ traits = { daoyu_target_character } } }
				else = { create_character = { age = 35 is_agitator = yes noble = no interest_group = $IG$ traits = { daoyu_target_character } } }
			}
			else_if = {
				limit = { has_global_variable = daoyu_create_ruler }
				hidden_effect = { every_scope_character = { limit = { or = { is_ruler = yes is_heir = yes } } kill_character = { hidden = yes } } }
				if = { limit = { has_global_variable = daoyu_noble } create_character = { age = 35 ruler = yes ig_leader = yes noble = yes interest_group = $IG$ traits = { daoyu_target_character } } }
				else = { create_character = { age = 35 ruler = yes ig_leader = yes noble = no interest_group = $IG$ traits = { daoyu_target_character } } }
			}
			else_if = {
				limit = { has_global_variable = daoyu_create_heir }
				hidden_effect = { random_scope_character = { limit = { is_heir = yes } kill_character = { hidden = yes } } }
				if = {
					limit = { any_scope_character = { is_ruler = yes age >= 35 } }
					if = { limit = { has_global_variable = daoyu_noble } create_character = { age = 16 heir = yes noble = yes interest_group = $IG$ traits = { daoyu_target_character } } }
					else = { create_character = { age = 16 heir = yes noble = no interest_group = $IG$ traits = { daoyu_target_character } } }
				}
				if = {
					limit = { any_scope_character = { is_ruler = yes age >= 19 age < 35 } }
					if = { limit = { has_global_variable = daoyu_noble } create_character = { age = 0 heir = yes noble = yes interest_group = $IG$ traits = { daoyu_target_character trait_child } } }
					else = { create_character = { age = 0 heir = yes noble = no interest_group = $IG$ traits = { daoyu_target_character trait_child } } }
				}
			}
			random_scope_character = { limit = { has_trait = daoyu_target_character } save_scope_as = daoyu_target_character }
		}
		else = {
			if = {
				limit = { has_global_variable = daoyu_create_politician }
				if = { limit = { has_global_variable = daoyu_noble } create_character = { age = 28 female = yes ig_leader = yes noble = yes interest_group = $IG$ traits = { daoyu_target_character } } }
				else = { create_character = { age = 28 female = yes ig_leader = yes noble = no interest_group = $IG$ traits = { daoyu_target_character } } }
			}
			else_if = {
				limit = { has_global_variable = daoyu_create_agitator }
				if = { limit = { has_global_variable = daoyu_noble } create_character = { age = 28 female = yes is_agitator = yes noble = yes interest_group = $IG$ traits = { daoyu_target_character } } }
				else = { create_character = { age = 28 female = yes is_agitator = yes noble = no interest_group = $IG$ traits = { daoyu_target_character } } }
			}
			else_if = {
				limit = { has_global_variable = daoyu_create_ruler }
				hidden_effect = { every_scope_character = { limit = { or = { is_ruler = yes is_heir = yes } } kill_character = { hidden = yes } } }
				if = {
					limit = { coa_monarchy_trigger = yes }
					if = { limit = { has_global_variable = daoyu_noble } create_character = { age = 18 female = yes ruler = yes ig_leader = yes noble = yes interest_group = $IG$ traits = { daoyu_target_character } } }
					else = { create_character = { age = 18 female = yes ruler = yes ig_leader = yes noble = no interest_group = $IG$ traits = { daoyu_target_character } } }
				}
				else = {
					if = { limit = { has_global_variable = daoyu_noble } create_character = { age = 35 female = yes ruler = yes ig_leader = yes noble = yes interest_group = $IG$ traits = { daoyu_target_character } } }
					else = { create_character = { age = 35 female = yes ruler = yes ig_leader = yes noble = no interest_group = $IG$ traits = { daoyu_target_character } } }
				}
			}
			else_if = {
				limit = { has_global_variable = daoyu_create_heir }
				hidden_effect = { random_scope_character = { limit = { is_heir = yes } kill_character = { hidden = yes } } }
				if = {
					limit = { any_scope_character = { is_ruler = yes age >= 35 } }
					if = { limit = { has_global_variable = daoyu_noble } create_character = { age = 16 female = yes heir = yes noble = yes interest_group = $IG$ traits = { daoyu_target_character } } }
					else = { create_character = { age = 16 female = yes heir = yes noble = no interest_group = $IG$ traits = { daoyu_target_character } } }
				}
				if = {
					limit = { any_scope_character = { is_ruler = yes age >= 19 age < 35 } }
					if = { limit = { has_global_variable = daoyu_noble } create_character = { age = 0 female = yes heir = yes noble = yes interest_group = $IG$ traits = { daoyu_target_character trait_child } } }
					else = { create_character = { age = 0 female = yes heir = yes noble = no interest_group = $IG$ traits = { daoyu_target_character trait_child } } }
				}
			}
			random_scope_character = { limit = { has_trait = daoyu_target_character } save_scope_as = daoyu_target_character }
		}
		remove_global_variable = daoyu_noble
	}
}
