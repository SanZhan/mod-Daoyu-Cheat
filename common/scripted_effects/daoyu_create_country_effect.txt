	set_global_variable = daoyu_count_tag_effect
	set_variable = { name = daoyu_count value = 0 }
	daoyu_all_tag_effect = yes
	remove_global_variable = daoyu_count_tag_effect

	set_global_variable = daoyu_remove_show_tag_effect
	daoyu_all_tag_effect = yes
	remove_global_variable = daoyu_remove_show_tag_effect
}
daoyu_single_tag_effect = {
	if = {
		limit = { has_global_variable = daoyu_remove_show_tag_effect }
		if = { limit = { has_variable = daoyu_show_tag_$TAG$ } remove_variable = daoyu_show_tag_$TAG$ }
	}
	if = {
		limit = { has_global_variable = daoyu_count_tag_effect }
		if = {
			limit = { daoyu_show_nonexistent_country_$TAG$ = yes }
			change_variable = { name = daoyu_count add = 1 }
			if = { limit = { has_variable = daoyu_page_01 has_variable = daoyu_count var:daoyu_count <= 14 } set_variable = daoyu_show_tag_$TAG$ }
			if = { limit = { has_variable = daoyu_page_02 has_variable = daoyu_count var:daoyu_count > 14 var:daoyu_count <= 27 } set_variable = daoyu_show_tag_$TAG$ }
			if = { limit = { has_variable = daoyu_page_03 has_variable = daoyu_count var:daoyu_count > 27 var:daoyu_count <= 40 } set_variable = daoyu_show_tag_$TAG$ }
			if = { limit = { has_variable = daoyu_page_04 has_variable = daoyu_count var:daoyu_count > 40 var:daoyu_count <= 53 } set_variable = daoyu_show_tag_$TAG$ }
			if = { limit = { has_variable = daoyu_page_05 has_variable = daoyu_count var:daoyu_count > 53 var:daoyu_count <= 66 } set_variable = daoyu_show_tag_$TAG$ }
			if = { limit = { has_variable = daoyu_page_06 has_variable = daoyu_count var:daoyu_count > 66 } set_variable = daoyu_show_tag_$TAG$ }
		}
	}
