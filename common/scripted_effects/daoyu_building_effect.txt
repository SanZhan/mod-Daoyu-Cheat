daoyu_current_building_after_single_effect = {
	if = {
		limit = { has_global_variable = daoyu_get_building_level_$ORDER$ }
		if = {
			limit = { has_global_variable = daoyu_buildings_expand }
			daoyu_get_building_level = yes
			change_variable = { name = daoyu_current_building_$ORDER$ add = 1 }
			set_global_variable = daoyu_current_buildings_expand_$ORDER$
			daoyu_current_buildings_expand = yes
		}
		if = {
			limit = { or = { has_global_variable = daoyu_ucy_nationally_owned has_global_variable = daoyu_nationalized_buildings } }
			daoyu_get_building_level = yes
			set_variable = daoyu_change_option_$ORDER$
			scope:daoyu_current_building_$ORDER$ = { add_modifier = { name = daoyu_current_building days = 1 } } daoyu_num_effect = yes
		}
	}
