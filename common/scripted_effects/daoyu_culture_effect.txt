daoyu_culture_effect = {
	if = {
		limit = { has_global_variable = daoyu_target_character_mark_target_culture }
		scope:daoyu_target_character = { change_character_culture = cu:$CULTURE$ }
	}
	if = {
		limit = { has_global_variable = daoyu_one_earth_one_nation_one_religion }
		cu:$CULTURE$ = { save_scope_as = daoyu_only_one_culture }
	}
	if = {
		limit = { has_global_variable = daoyu_new_culture_for_name }
		cu:$CULTURE$ = { save_scope_as = daoyu_new_culture_for_name }
	}
	if = {
		limit = { has_global_variable = daoyu_mark_target_culture_begin }
		set_global_variable = daoyu_mark_target_culture_complete set_global_variable = dculo_$CULTURE$
		remove_global_variable = daoyu_mark_target_culture_begin
	}
	if = {
		limit = { has_global_variable = daoyu_r_cu exists = scope:daoyu_target_country }
		scope:daoyu_target_country = { remove_primary_culture = cu:$CULTURE$ }
	}
	if = {
		limit = { has_global_variable = daoyu_a_cu exists = scope:daoyu_target_country }
		scope:daoyu_target_country = { add_primary_culture = cu:$CULTURE$ }
	}
	if = {
		limit = { has_global_variable = daoyu_m_cu exists = scope:daoyu_target_country exists = scope:daoyu_target_state }
		scope:daoyu_target_state = {
			limit = { state_is_eligible_as_mass_migration_target = yes }
			if = { limit = { not = { has_modifier = daoyu_migration_pull } } add_modifier = { name = daoyu_migration_pull multiplier = 100 years = 2 } }
			remove_variable = daoyu_target_state
			create_mass_migration = { origin = scope:daoyu_target_country culture = cu:$CULTURE$ }
		}
	}
	if = {
		limit = { has_global_variable = daoyu_c_cu exists = scope:daoyu_target_country }
		scope:daoyu_target_country = {
			every_scope_state = {
				if = { limit = { has_cultural_community = cu:$CULTURE$ not = { has_cultural_community = scope:daoyu_t_cu } } add_cultural_community = scope:daoyu_t_cu }
				if = { limit = { state_region = { is_homeland = cu:$CULTURE$ } not = { state_region = { is_homeland = scope:daoyu_t_cu } } } state_region = { add_homeland = scope:daoyu_t_cu } }
				if = { limit = { state_region = { is_homeland = cu:$CULTURE$ not = { any_scope_state = { not = { owner = scope:daoyu_target_country } any_scope_pop = { has_pop_culture = $CULTURE$ } } } } } state_region = { remove_homeland = cu:$CULTURE$ } }
				every_scope_pop = { limit = { has_pop_culture = $CULTURE$ } change_pop_culture = { value = 1 target = scope:daoyu_t_cu } }
			}
		}
	}
	if = {
		limit = { has_global_variable = daoyu_remove_homeland }
		every_scope_state = { limit = { has_variable = daoyu_target_state } state_region = { remove_homeland = cu:$CULTURE$ } }
	}
	if = {
		limit = { has_global_variable = daoyu_add_homeland }
		every_scope_state = { limit = { has_variable = daoyu_target_state } state_region = { add_homeland = cu:$CULTURE$ } }
	}
}
daoyu_count_culture_effect = {
	set_global_variable = daoyu_count_culture_effect
	if = { limit = { not = { has_variable = daoyu_count } } set_variable = { name = daoyu_count value = 0 } }
	daoyu_all_culture_effect = yes
	remove_global_variable = daoyu_count_culture_effect
}

}
daoyu_change_target_character_culture = {
	if = {
		limit = { not = { exists = scope:daoyu_target_character } }
		random_character = { limit = { has_trait = daoyu_target_character } save_scope_as = daoyu_target_character }
	}
	set_global_variable = daoyu_change_target_character_culture
	daoyu_all_culture_effect = yes
}


daoyu_single_culture_effect = {
	if = {
		limit = { has_global_variable = daoyu_remove_show_cu_effect }
		if = { limit = { has_variable = daoyu_show_cu_$CULTURE$ } remove_variable = daoyu_show_cu_$CULTURE$ }
	}
	if = {
		limit = { has_global_variable = daoyu_save_primary_culture_effect }
		if = { limit = { exists = scope:daoyu_target_country scope:daoyu_target_country = { country_has_primary_culture = cu:$CULTURE$ } } cu:$CULTURE$ = { save_scope_as = daoyu_t_cu } }
	}
	if = {
		limit = { has_global_variable = daoyu_count_culture_effect }
		if = {
			limit = { has_global_variable = daoyu_add_homeland }
			if = {
				limit = { any_scope_state = { has_variable = daoyu_target_state any_scope_pop = { has_pop_culture = $CULTURE$ } not = { state_region = { is_homeland = cu:$CULTURE$ } } } }
				change_variable = { name = daoyu_count add = 1 }
				if = { limit = { has_variable = daoyu_page_01 has_variable = daoyu_count var:daoyu_count <= 14 } set_variable = daoyu_show_cu_$CULTURE$ }
				if = { limit = { has_variable = daoyu_page_02 has_variable = daoyu_count var:daoyu_count > 14 var:daoyu_count <= 27 } set_variable = daoyu_show_cu_$CULTURE$ }
				if = { limit = { has_variable = daoyu_page_03 has_variable = daoyu_count var:daoyu_count > 27 } set_variable = daoyu_show_cu_$CULTURE$ }
			}
		}
		if = {
			limit = { has_global_variable = daoyu_remove_homeland }
			if = {
				limit = { any_scope_state = { has_variable = daoyu_target_state state_region = { is_homeland = cu:$CULTURE$ } } }
				change_variable = { name = daoyu_count add = 1 }
				if = { limit = { has_variable = daoyu_page_01 has_variable = daoyu_count var:daoyu_count <= 14 } set_variable = daoyu_show_cu_$CULTURE$ }
				if = { limit = { has_variable = daoyu_page_02 has_variable = daoyu_count var:daoyu_count > 14 var:daoyu_count <= 27 } set_variable = daoyu_show_cu_$CULTURE$ }
				if = { limit = { has_variable = daoyu_page_03 has_variable = daoyu_count var:daoyu_count > 27 } set_variable = daoyu_show_cu_$CULTURE$ }
			}
		}
		if = {
			limit = { or = { has_global_variable = daoyu_mark_target_culture_begin has_global_variable = daoyu_one_earth_one_nation_one_religion } }
			if = {
				limit = { daoyu_cu_$CULTURE$ = yes has_variable = daoyu_count }
				change_variable = { name = daoyu_count add = 1 }
				if = { limit = { has_variable = daoyu_page_01 has_variable = daoyu_count var:daoyu_count <= 14 } set_variable = daoyu_show_cu_$CULTURE$ }
				if = { limit = { has_variable = daoyu_page_02 has_variable = daoyu_count var:daoyu_count > 14 var:daoyu_count <= 27 } set_variable = daoyu_show_cu_$CULTURE$ }
				if = { limit = { has_variable = daoyu_page_03 has_variable = daoyu_count var:daoyu_count > 27 } set_variable = daoyu_show_cu_$CULTURE$ }
			}
		}
		if = {
			limit = { has_global_variable = daoyu_m_cu exists = scope:daoyu_target_country }
			if = {
				limit = { scope:daoyu_target_country = { any_scope_state = { any_scope_pop = { has_pop_culture = $CULTURE$ } } } }
				change_variable = { name = daoyu_count add = 1 }
				if = { limit = { has_variable = daoyu_page_01 has_variable = daoyu_count var:daoyu_count <= 14 } set_variable = daoyu_show_cu_$CULTURE$ }
				if = { limit = { has_variable = daoyu_page_02 has_variable = daoyu_count var:daoyu_count > 14 var:daoyu_count <= 27 } set_variable = daoyu_show_cu_$CULTURE$ }
				if = { limit = { has_variable = daoyu_page_03 has_variable = daoyu_count var:daoyu_count > 27 } set_variable = daoyu_show_cu_$CULTURE$ }
			}
		}
		if = {
			limit = { has_global_variable = daoyu_r_cu exists = scope:daoyu_target_country }
			if = {
				limit = { scope:daoyu_target_country = { country_has_primary_culture = cu:$CULTURE$ } }
				change_variable = { name = daoyu_count add = 1 }
				if = { limit = { has_variable = daoyu_page_01 has_variable = daoyu_count var:daoyu_count <= 14 } set_variable = daoyu_show_cu_$CULTURE$ }
				if = { limit = { has_variable = daoyu_page_02 has_variable = daoyu_count var:daoyu_count > 14 var:daoyu_count <= 27 } set_variable = daoyu_show_cu_$CULTURE$ }
				if = { limit = { has_variable = daoyu_page_03 has_variable = daoyu_count var:daoyu_count > 27 } set_variable = daoyu_show_cu_$CULTURE$ }
			}
		}
		if = {
			limit = { or = { has_global_variable = daoyu_a_cu has_global_variable = daoyu_c_cu } exists = scope:daoyu_target_country }
			if = {
				limit = { scope:daoyu_target_country = { any_scope_state = { any_scope_pop = { pop_has_primary_culture = no has_pop_culture = $CULTURE$ } } } }
				change_variable = { name = daoyu_count add = 1 }
				if = { limit = { has_variable = daoyu_page_01 has_variable = daoyu_count var:daoyu_count <= 14 } set_variable = daoyu_show_cu_$CULTURE$ }
				if = { limit = { has_variable = daoyu_page_02 has_variable = daoyu_count var:daoyu_count > 14 var:daoyu_count <= 27 } set_variable = daoyu_show_cu_$CULTURE$ }
				if = { limit = { has_variable = daoyu_page_03 has_variable = daoyu_count var:daoyu_count > 27 } set_variable = daoyu_show_cu_$CULTURE$ }
			}
		}
	}
	if = {
		limit = { has_global_variable = daoyu_remove_homeland_effect }
		every_state = {
			limit = { state_region = { is_homeland = cu:$CULTURE$ } }
			state_region = { remove_homeland = cu:$CULTURE$ }
		}
	}
	if = {
		limit = { has_global_variable = daoyu_add_radicals_in_state_effect }
		if = { limit = { owner = { country_has_primary_culture = cu:$CULTURE$ } state_region = { is_homeland = cu:$CULTURE$ } any_scope_pop = { strata = poor } } add_radicals_in_state = { value = 1 strata = poor } order_by = turmoil }
		if = {
			limit = { owner = { not = { country_has_primary_culture = cu:$CULTURE$ } } state_region = { is_homeland = cu:$CULTURE$ } any_scope_pop = { culture = cu:$CULTURE$ } }
			add_radicals_in_state = { value = 1 culture = cu:$CULTURE$ } order_by = turmoil 
			if = { limit = { has_global_variable = daoyu_country_add_radicals } add_culture_standard_of_living_modifier = { culture = cu:$CULTURE$ months = 1 multiplier = -100 } }
			else = { add_culture_standard_of_living_modifier = { culture = cu:$CULTURE$ months = 12 multiplier = -100 } }
		}
	}
	if = {
		limit = { has_global_variable = daoyu_add_homeland_for_mark_target_culture_effect }
		if = {
			limit = { has_global_variable = dculo_$CULTURE$ }
			every_state = {
				limit = { has_variable = daoyu_target_state state_region = { not = { is_homeland = cu:$CULTURE$ } } }
				state_region = { add_homeland = cu:$CULTURE$ }
				if = { limit = { not = { has_cultural_community = cu:$CULTURE$ } } add_cultural_community = cu:$CULTURE$ }
			}
		}
	}
	if = {
		limit = { has_global_variable = daoyu_add_owner_homeland_remove_others_effect }
		if = {
			limit = { owner = { country_has_primary_culture = cu:$CULTURE$ } }
			if = { limit = { not = { state_region = { is_homeland = cu:$CULTURE$ } } } state_region = { add_homeland = cu:$CULTURE$ } }
			if = { limit = { not = { has_cultural_community = cu:$CULTURE$ } } add_cultural_community = cu:$CULTURE$ }
		}
		if = {
			limit = { owner = { not = { country_has_primary_culture = cu:$CULTURE$ } } }
			owner = { save_temporary_scope_as = daoyu_temporary_owner }
			if = { limit = { state_region = { is_homeland = cu:$CULTURE$ not = { any_scope_state = { not = { owner = scope:daoyu_temporary_owner } any_scope_pop = { has_pop_culture = $CULTURE$ } } } } } state_region = { remove_homeland = cu:$CULTURE$ } }
		}
	}
	if = {
		limit = { has_global_variable = daoyu_create_pop_for_mark_target_culture_effect }
		if = {
			limit = { has_global_variable = dculo_$CULTURE$ }
			every_state = {
				limit = { has_variable = daoyu_target_state }
				every_scope_pop = {
					limit = { culture = cu:daoyu_nonexistent }
					if = { limit = { has_global_variable = daoyu_set_create_pops_L_R_02 } set_pop_literacy = { literacy_rate = 0.2 } }
					if = { limit = { has_global_variable = daoyu_set_create_pops_L_R_04 } set_pop_literacy = { literacy_rate = 0.4 } }
					if = { limit = { has_global_variable = daoyu_set_create_pops_L_R_06 } set_pop_literacy = { literacy_rate = 0.6 } }
					if = { limit = { has_global_variable = daoyu_set_create_pops_L_R_08 } set_pop_literacy = { literacy_rate = 0.8 } }
					if = { limit = { has_global_variable = daoyu_set_create_pops_L_R_10 } set_pop_literacy = { literacy_rate = 1 } }
					change_pop_culture = { value = 1 target = cu:$CULTURE$ }
				}
				if = { limit = { not = { has_cultural_community = cu:$CULTURE$ } } add_cultural_community = cu:$CULTURE$ }
			}
		}
	}
	if = {
		limit = { has_global_variable = daoyu_create_primary_culture_pops_effect }
		every_scope_state = {
			if = {
				limit = { root = { country_has_primary_culture = cu:$CULTURE$ } }
				every_scope_pop = {
					limit = { culture = cu:daoyu_nonexistent }
					if = { limit = { has_global_variable = daoyu_set_create_pops_L_R_02 } set_pop_literacy = { literacy_rate = 0.2 } }
					if = { limit = { has_global_variable = daoyu_set_create_pops_L_R_04 } set_pop_literacy = { literacy_rate = 0.4 } }
					if = { limit = { has_global_variable = daoyu_set_create_pops_L_R_06 } set_pop_literacy = { literacy_rate = 0.6 } }
					if = { limit = { has_global_variable = daoyu_set_create_pops_L_R_08 } set_pop_literacy = { literacy_rate = 0.8 } }
					if = { limit = { has_global_variable = daoyu_set_create_pops_L_R_10 } set_pop_literacy = { literacy_rate = 1.0 } }
					change_pop_culture = { value = 1 target = cu:$CULTURE$ }
				}
				if = { limit = { not = { has_cultural_community = cu:$CULTURE$ } } add_cultural_community = cu:$CULTURE$ }
			}
		}
	}
	if = {
		limit = { has_global_variable = dculo_remove_global_variable_effect }
		remove_global_variable = dculo_$CULTURE$
	}
	if = {
		limit = { has_global_variable = daoyu_save_mark_target_culture_effect }
		if = { limit = { has_global_variable = dculo_$CULTURE$ } cu:$CULTURE$ = { save_scope_as = daoyu_mark_target_culture } }
	}
	if = {
		limit = { has_global_variable = daoyu_remove_all_primary_culture_effect }
		remove_primary_culture = cu:$CULTURE$
	}
	if = {
		limit = { has_global_variable = daoyu_change_target_character_culture exists = scope:daoyu_target_character }
		if = {
			limit = { country_has_primary_culture = cu:$CULTURE$ }
			scope:daoyu_target_character = { change_character_culture = cu:$CULTURE$ }
			remove_global_variable = daoyu_change_target_character_culture
			clear_saved_scope = daoyu_target_character
		}
	}
}

	every_scope_state = {
		limit = { has_variable = daoyu_target_state }
