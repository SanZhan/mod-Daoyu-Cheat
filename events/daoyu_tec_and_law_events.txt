	immediate = {
		clear_saved_scope = daoyu_target_country
		random_state = { limit = { owner = { is_ai = yes has_modifier = daoyu_target_country } } owner = { save_scope_as = daoyu_target_country } }
	}
	option = {
		name = RELOAD#重新加载
		hidden_effect = { trigger_event = { id = d_law_tec.700 popup = yes } }

		name = daoyu_remove_target_country#重新选目标

		clear_saved_scope = daoyu_target_country
		hidden_effect = { trigger_event = { id = d_law_tec.700 popup = yes } }
	}
	option = {
		name = daoyu_modify_target_country#调整目标国家
		trigger = { exists = scope:daoyu_target_country scope:daoyu_target_country = { is_ai = yes has_modifier = daoyu_target_country } }
		hidden_effect = { trigger_event = { id = d_law_tec.720 popup = yes } }
	}
	option = {
		name = daoyu_modify_our_country#调整本国
		random_country = { limit = { is_ai = yes has_modifier = daoyu_target_country } remove_modifier = daoyu_target_country }
		add_modifier = { name = daoyu_target_country days = 1 }
		hidden_effect = { trigger_event = { id = d_law_tec.720 popup = yes } }
	}
	option = {
		name = RESIGN
		ai_chance = { base = 100 }
		hidden_effect = { trigger_event = { id = daoyu.10 popup = yes } }
	}
